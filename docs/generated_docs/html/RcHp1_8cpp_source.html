82 void RcHp1::calculateFilterCoefficients() {
83  // Clear coefficient array
84  memset(&coefficientsArray[0], 0, sizeof(double) * numCoefficients);
85 
86  // Calculate coefficients
87  const double w0 = 2.0 * PI * params.fc;
88  const double gamma = 1.0 / sampleRate * w0;
89 
90  // Update coefficient array
91  coefficientsArray[a0] = 2.0 / (gamma + 2.0);
92  coefficientsArray[a1] = -2.0 / (gamma + 2.0);
93  coefficientsArray[a2] = 0.0;
94  coefficientsArray[b1] = (gamma - 2.0) / (gamma + 2.0);
95  coefficientsArray[b2] = 0.0;
96 
97  biquad.setCoefficients(coefficientsArray);
98 }
