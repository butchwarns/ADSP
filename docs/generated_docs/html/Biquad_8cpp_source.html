35 double Biquad::process(double x) {
36  switch (parameters.calculationType) {
37  // Direct form
38  case biquadAlgorithm::direct: {
39  // y[n] = a0*x[n] + a1*x[n-1] + a2*x[n-2] - b1*y[n-1] - b2*y[n-2]
40  double y = coefficientsArray[a0] * x +
41  coefficientsArray[a1] * stateArray[x_z1] +
42  coefficientsArray[a2] * stateArray[x_z2] -
43  coefficientsArray[b1] * stateArray[y_z1] -
44  coefficientsArray[b2] * stateArray[y_z2];
45 
46  fixUnderflow(y);
47 
48  // Update state registers
49  stateArray[x_z2] = stateArray[x_z1];
50  stateArray[x_z1] = x;
51 
52  stateArray[y_z2] = stateArray[y_z1];
53  stateArray[y_z1] = y;
54 
55  // Output
56  return y;
57  }
58 
59  // Canonical form, uses only two state registers
60  case biquadAlgorithm::canonical: {
61  // w[n] = x[n] - b1*w[n-1] - b2*w[n-2]
62  double w = x - coefficientsArray[b1] * stateArray[x_z1] -
63  coefficientsArray[b2] * stateArray[x_z2];
64 
65  // y[n] = a0*w[n] + a1*w[n-1] + a2*w[n-2]
66  double y = coefficientsArray[a0] * w +
67  coefficientsArray[a1] * stateArray[x_z1] +
68  coefficientsArray[a2] * stateArray[x_z2];
69 
70  fixUnderflow(y);
71 
72  // Update state registers
73  stateArray[x_z2] = stateArray[x_z1];
74  stateArray[x_z1] = w;
75 
76  // Output
77  return y;
78  }
79 
80  // Transposed direct form
81  case biquadAlgorithm::transposedDirect: {
82  // w[n] = x[n] + stateArray[y_z1]
83  double w = x + stateArray[y_z1];
84  // y[n] = a0*w[n] + stateArray[x_z1]
85  double y = coefficientsArray[a0] * w + stateArray[x_z1];
86 
87  fixUnderflow(y);
88 
89  // Update state registers
90  stateArray[y_z1] = stateArray[y_z2] - coefficientsArray[b1] * w;
91  stateArray[y_z2] = -coefficientsArray[b2] * w;
92 
93  stateArray[x_z1] = stateArray[x_z2] + coefficientsArray[a1] * w;
94  stateArray[x_z2] = coefficientsArray[a2] * w;
95 
96  // Output
97  return y;
98  }
99 
100  // Transposed canonical form
101  case biquadAlgorithm::transposedCanonical: {
102  // y[n] = a0*x[n] + stateArray[x_z1]
103  double y = coefficientsArray[a0] * x + stateArray[x_z1];
104 
105  fixUnderflow(y);
106 
107  // Update state registers
108  stateArray[x_z1] = coefficientsArray[a1] * x -
109  coefficientsArray[b1] * y + stateArray[x_z2];
110 
111  stateArray[x_z2] =
112  coefficientsArray[a2] * x - coefficientsArray[b2] * y;
113 
114  // Output
115  return y;
116  }
117 
118  default: {
119  return x; // Did not process sample
120  }
121  }
122 }
