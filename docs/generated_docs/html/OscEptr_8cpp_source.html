58 void OscEptr::setParams(const OscEptrParams &_params) {
59  // Only if new params differ..
60  if (params.frequency != _params.frequency ||
61  params.pulseWidth != _params.pulseWidth) {
62  // Update params
63  params = _params;
64 
65  // Calculate delta according to new frequency
66  delta = _params.frequency * T;
67 
68  // Calculate period length
69  period = 1.0 / params.frequency;
70 
71  // Find gradients for asymmetric triangle
72  // gradients of 2.0 give regular triangle wave
73  gradientUp = 1.0 / params.pulseWidth;
74  gradientDown = -gradientUp / (gradientUp - 1.0);
75 
76  // Correction coefficients for asymmetric TRI
77  corrCoeffs[eptrCorrCoeffs::a0] =
78  -1.0 * ((gradientUp * delta - 1.0) * (gradientUp * delta - 1.0)) /
79  (4.0 * delta * (gradientUp - 1.0));
80  corrCoeffs[eptrCorrCoeffs::a1] =
81  (2.0 * gradientUp * delta - 4.0 * delta + 2.0) /
82  (4.0 * delta * (gradientUp - 1.0));
83  corrCoeffs[eptrCorrCoeffs::a2] =
84  -1.0 / (4.0 * delta * (gradientUp - 1.0));
85 
86  corrCoeffs[eptrCorrCoeffs::b0] = -1.0 * (gradientDown * delta + 1.0) *
87  (gradientDown * delta + 1.0) /
88  (4.0 * delta * (gradientDown + 1.0));
89  corrCoeffs[eptrCorrCoeffs::b1] =
90  (2.0 * gradientDown * delta + 4.0 * delta - 2.0) /
91  (4.0 * delta * (gradientDown + 1.0));
92  corrCoeffs[eptrCorrCoeffs::b2] =
93  -1.0 / (4.0 * delta * (gradientDown + 1.0));
94  }
95 }

105 double OscEptr::correctCounterTriMin() {
106  return corrCoeffs[eptrCorrCoeffs::b2] * counterTri * counterTri +
107  corrCoeffs[eptrCorrCoeffs::b1] * counterTri +

117 OscEptrOut OscEptr::renderOutputs() {
118  //==============================================================================
119  // SAW
120 
121  double saw{0.0};
122 
123  counterSaw += 2.0 * delta;
124  if (counterSaw > (1.0 - delta)) {
125  saw = correctCounterSaw();
126  counterSaw -= 2.0;
127  } else {
128  saw = counterSaw;
129  }
130 
131  //==============================================================================
132  // TRIANGLE
133 
134  double tri{0.0};
135 
136  // Upwards
137  if (dir == oscDirection::UP) {
138  counterTri += 2.0 * gradientUp * delta;
139  // Transition
140  if (counterTri > 1.0 - gradientUp * delta) {
141  tri = correctCounterTriMax();
142  counterTri = 1.0 + (counterTri - 1.0) * gradientDown / gradientUp;
143  // Reverse direction
144  dir = oscDirection::DOWN;
145  }
146  // Linear section
147  else {
148  tri = counterTri;
149  }
150  }
151  // Downwards
152  else {
153  counterTri += 2.0 * gradientDown * delta;
154  // Transition
155  if (counterTri < -1.0 - gradientDown * delta) {
156  tri = correctCounterTriMin();
157  counterTri = -1.0 + (counterTri + 1.0) * gradientUp / gradientDown;
158  // Reverse direction
159  dir = oscDirection::UP;
160  }
161  // Linear section
162  else {
163  tri = counterTri;
164  }
165  }
166 
167  //==============================================================================
168  // PULSE
169 
170  double pulse = 0.0;
171 
172  // Upper half wave (triangle going upwards)
173  if (dir == oscDirection::UP) {
174  // Transition region
175  if (tri > 1.0 - 2.0 * gradientUp * T) {
176  // Correct discontinuity like in the saw case
177  pulse = correctCounterSaw();
178  } else {
179  pulse = 1.0;
180  }
181  }
182  // Lower half wave (triangle goind downwards)
183  else {
184  // Transition region
185  if (tri < -1.0 + 2.0 * gradientDown * T) {
186  // Correct discontinuity like in the saw case (use symmetry)
187  pulse = -1.0 * correctCounterSaw();
188  } else {
189  pulse = -1.0;
190  }
191  }
192 
193  //==============================================================================
194  // Compensate for dc offset resulting from changing pulse width
195 
196  // Pulse wave offset compensation
197  const double pulseOffset =
198  (params.pulseWidth - (1.0 - params.pulseWidth)) * period;
199  pulse -= pulseOffset;
200 
201  // Triangle wave offset comepnsation
202  const double triOffset = pulseOffset * 0.5;
203  tri -= triOffset;
204 
205  //==============================================================================
206  // Output
207 
208  OscEptrOut output;
209  output.saw = saw;
210  output.tri = tri;
211  output.pulse = pulse;
212 
213  return output;
214 }
