82 void SkHp2::calculateFilterCoefficients() {
83  // Clear coefficient array
84  memset(&coefficientsArray[0], 0, sizeof(double) * numCoefficients);
85 
86  // Calculate coefficients
87  const double w0 = 2.0 * PI * params.fc;
88 
89  const double alpha = 1.0 / sampleRate * w0;
90  const double alpha2 = alpha * alpha;
91 
92  const double muDen = alpha2 + 4.0 * alpha + 4.0;
93  const double mu = 4.0 / muDen;
94 
95  // Update coefficient array
96  coefficientsArray[a0] = mu;
97  coefficientsArray[a1] = -2.0 * mu;
98  coefficientsArray[a2] = mu;
99  coefficientsArray[b1] = (2.0 * alpha - 4.0) / (alpha + 2.0);
100  coefficientsArray[b2] = (alpha2 - 4.0 * alpha + 4.0) / muDen;
101 
102  biquad.setCoefficients(coefficientsArray);
103 }
