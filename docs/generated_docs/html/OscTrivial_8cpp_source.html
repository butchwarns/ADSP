64 OscTrivialOut OscTrivial::renderOutputs() {
65  //==============================================================================
66  // SAW
67 
68  // Bipolar modulo counter
69  counter += delta;
70  wrap(counter);
71  const double saw = unipolarToBipolar(counter);
72 
73  //==============================================================================
74  // TRIANGLE
75 
76  // Rectify, scale and offset saw to create triangle
77  // Offset counter to align phases between waveforms
78  counterTri += delta;
79  wrap(counterTri);
80  const double sawTri = unipolarToBipolar(counterTri);
81  const double tri = unipolarToBipolar(fabs(sawTri));
82 
83  //==============================================================================
84  // PULSE
85 
86  const double pulse = (saw < 0.0) ? 1.0 : -1.0;
87 
88  //==============================================================================
89  // Output
90 
91  OscTrivialOut output;
92  output.saw = saw;
93  output.tri = tri;
94  output.pulse = pulse;
95 
96  return output;
97 }

99 void OscTrivial::retrigger() {
100  // Reset counter to initial phase angle
101  counter = params.phase;
102 
103  // Set triangle counter to correct initial phase (with alignment to other waveforms)
104  double phaseTri = params.phase + 0.5;
105  wrap(phaseTri);
106  counterTri = phaseTri;
107 }
